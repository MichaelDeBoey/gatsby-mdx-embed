import { Playground, Props } from "docz"
import { TwitterFollowButton } from "@pauliescanlon/gatsby-mdx-embed"

# TwitterFollowButton

Display a TwitterFollowButton by including the component in your `.mdx`. The `username` prop is required.

<Props name="TwitterFollowButton" of={TwitterFollowButton} />

## Basic usage

<Playground>
  <TwitterFollowButton username="PaulieScanlon" />
</Playground>

## Show followers

<Playground>
  <TwitterFollowButton username="gatsbyjs" showFollowers={true} />
</Playground>

## Show username

<Playground>
  <TwitterFollowButton username="PaulieScanlon" showUsername={false} />
</Playground>

## Size

<Playground>
  <TwitterFollowButton username="PaulieScanlon" size="large" />
</Playground>
