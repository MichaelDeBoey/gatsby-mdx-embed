import { Playground, Props } from "docz"
import { Tweet } from "@pauliescanlon/gatsby-mdx-embed"

# Tweet

Display a Tweet by including the component in your `.mdx`. The `tweetLink` prop is required.

<Props name="Tweet" of={Tweet} />

## Basic usage

<Playground>
  <Tweet tweetLink="PaulieScanlon/status/1232982448310497286" />
</Playground>

## Theme

<Playground>
  <Tweet tweetLink="PaulieScanlon/status/1232982448310497286" theme="dark" />
</Playground>

## Align

<Playground>
  <Tweet tweetLink="PaulieScanlon/status/1232982448310497286" align="right" />
</Playground>

## Hide Conversation

<Playground>
  <Tweet
    tweetLink="https://twitter.com/geekgalgroks/status/1255864483739729922"
    hideConversation={true}
  />
</Playground>
